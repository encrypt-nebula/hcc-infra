✅ 3. Python Example – Detect MEAT
def detect_meat(sentence):
   sentence = sentence.lower()
   results = {}


   for meat, data in MEAT_DICT["categories"].items():
       results[meat] = any(k in sentence for k in data["keywords"])


   if any(n in sentence for n in MEAT_DICT["negative_keywords"]):
       return {"valid": False}


   return results

# -----------------------------

LABS = [
   "cbc", "cmp", "bmp", "lipid panel",
   "thyroid test", "psa", "troponin",
   "bnp", "a1c", "creatinine",
   "gfr", "liver enzymes"

# -----------------------------
# 4. Medication Keywords

MEDS = [

# -----------------------------
# 5. Generate Library
# -----------------------------
library = {}


for meat_type, actions in MEAT_ACTIONS.items():
   terms = []


   # Disease phrases
   for disease, keywords in DISEASES.items():
       combos = [
           f"{a} {k}" for a, k in product(actions, keywords)
       ]
       terms.extend(combos)


   # Lab phrases
   lab_combos = [
       f"{a} {lab}" for a, lab in product(actions, LABS)
   ]
   terms.extend(lab_combos)


   # Medication phrases
   med_combos = [
       f"{a} {med}" for a, med in product(actions, MEDS)
   ]
   terms.extend(med_combos)


   library[meat_type] = list(set(terms))


# -----------------------------
# 6. Negative Keywords
# -----------------------------
NEGATIVE = [
   "history of", "family history",
   "rule out", "possible",
   "suspected", "screening only",
   "denies", "no evidence of"
]


output = {
   "version": "1.0",
   "total_terms_estimated": sum(len(v) for v in library.values()),
   "categories": library,
   "negative_keywords": NEGATIVE
}


with open("medryte_10000_keyword_library.json", "w") as f:
   json.dump(output, f, indent=2)
